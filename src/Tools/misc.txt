/* ── findPattern ─────────────────────────────────────────────────────────── */

uintptr_t findPattern(const std::uint8_t *regionBase,
                      std::size_t regionSize,
                      const std::uint8_t *pattern,
                      const char *mask)
{
    if (!regionBase || !pattern || !mask || regionSize == 0)
        return 0;

    const std::size_t patLen = std::strlen(mask);
    if (patLen == 0 || patLen > regionSize)
        return 0;

    const std::size_t limit = regionSize - patLen;
    for (std::size_t off = 0; off <= limit; ++off) {
        bool ok = true;
        for (std::size_t i = 0; i < patLen; ++i) {
            if (mask[i] == 'x' && regionBase[off + i] != pattern[i]) {
                ok = false;
                break;
            }
        }
        if (ok) return reinterpret_cast<uintptr_t>(regionBase + off);
    }
    return 0;
}
// Use std::string_view for the mask to get size for free
uintptr_t findPattern_alt(const std::uint8_t *regionBase, std::size_t regionSize,
    const std::uint8_t *pattern, std::string_view mask) {
    const std::size_t patLen = mask.length();
    if (!regionBase || regionSize < patLen) return 0;

    const std::size_t limit = regionSize - patLen;
    for (std::size_t off = 0; off <= limit; ++off) {
        bool ok = true;
        for (std::size_t i = 0; i < patLen; ++i) {
            // '?' is often used for wildcards in these masks
            if (mask[i] == 'x' && regionBase[off + i] != pattern[i]) {
                ok = false;
                break;
            }
        }
        if (ok) return reinterpret_cast<uintptr_t>(regionBase + off);
    }
    return 0;
}














bool Inject::injectWithShellcode(DWORD targetPid)
{
    if (m_dllPath.empty() || targetPid == 0) return false;

    // 1. Open target with precise access rights
    HANDLE hProc = OpenProcess(PROCESS_CREATE_THREAD | PROCESS_QUERY_INFORMATION | 
                              PROCESS_VM_OPERATION | PROCESS_VM_WRITE | PROCESS_VM_READ, 
                              FALSE, targetPid);
    if (!hProc) return false;

    // 2. Determine if Target is x86 or x64
    BOOL isTargetWow64 = FALSE;
    IsWow64Process(hProc, &isTargetWow64);
    
    // Check if OUR process bitness matches the target
    // Note: Cross-bitness injection (x64 -> x86) usually requires advanced techniques
    // like "Heaven's Gate," but for a standard library, we match architectures.
    
    // 3. Allocate and Write the DLL Path
    size_t pathLen = m_dllPath.size() + 1;
    LPVOID remotePath = VirtualAllocEx(hProc, nullptr, pathLen, MEM_COMMIT | MEM_RESERVE, PAGE_READWRITE);
    if (!remotePath) { CloseHandle(hProc); return false; }
    WriteProcessMemory(hProc, remotePath, m_dllPath.c_str(), pathLen, nullptr);

    // 4. Prepare Shellcode
    std::vector<unsigned char> finalShellcode;
    LPVOID loadLibAddr = (LPVOID)GetProcAddress(GetModuleHandleA("kernel32.dll"), "LoadLibraryA");

    if (isTargetWow64) {
        // --- x86 Shellcode (STDCALL) ---
        unsigned char x86Code[] = {
            0x68, 0x00, 0x00, 0x00, 0x00, // push remotePath
            0xB8, 0x00, 0x00, 0x00, 0x00, // mov eax, loadLibAddr
            0xFF, 0xD0,                   // call eax
            0xC2, 0x04, 0x00              // ret 4
        };
        *reinterpret_cast<uint32_t*>(&x86Code[1]) = static_cast<uint32_t>(reinterpret_cast<uintptr_t>(remotePath));
        *reinterpret_cast<uint32_t*>(&x86Code[6]) = static_cast<uint32_t>(reinterpret_cast<uintptr_t>(loadLibAddr));
        finalShellcode.assign(x86Code, x86Code + sizeof(x86Code));
    } 
    else {
        // --- x64 Shellcode (FastCall) ---
        unsigned char x64Code[] = {
            0x48, 0x83, 0xEC, 0x28,                                     // sub rsp, 28h (Shadow space)
            0x48, 0xB9, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, // mov rcx, remotePath
            0x48, 0xB8, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, // mov rax, loadLibAddr
            0xFF, 0xD0,                                                 // call rax
            0x48, 0x83, 0xC4, 0x28,                                     // add rsp, 28h
            0xC3                                                        // ret
        };
        *reinterpret_cast<LPVOID*>(&x64Code[6])  = remotePath;
        *reinterpret_cast<LPVOID*>(&x64Code[16]) = loadLibAddr;
        finalShellcode.assign(x64Code, x64Code + sizeof(x64Code));
    }

    // 5. Create Code Cave
    LPVOID remoteCode = VirtualAllocEx(hProc, nullptr, finalShellcode.size(), MEM_COMMIT | MEM_RESERVE, PAGE_EXECUTE_READWRITE);
    if (!remoteCode) {
        VirtualFreeEx(hProc, remotePath, 0, MEM_RELEASE);
        CloseHandle(hProc);
        return false;
    }
    
    WriteProcessMemory(hProc, remoteCode, finalShellcode.data(), finalShellcode.size(), nullptr);

    // 6. Execute
    HANDLE hThread = CreateRemoteThread(hProc, nullptr, 0, (LPTHREAD_START_ROUTINE)remoteCode, nullptr, 0, nullptr);
    
    bool success = false;
    if (hThread) {
        WaitForSingleObject(hThread, 5000); // 5-second timeout optimization
        DWORD exitCode = 0;
        GetExitCodeThread(hThread, &exitCode);
        success = (exitCode != 0); 
        CloseHandle(hThread);
    }

    // 7. Cleanup (Optional: In a "stealth" version, you might leave the cave)
    VirtualFreeEx(hProc, remoteCode, 0, MEM_RELEASE);
    VirtualFreeEx(hProc, remotePath, 0, MEM_RELEASE);
    CloseHandle(hProc);

    return success;
}
